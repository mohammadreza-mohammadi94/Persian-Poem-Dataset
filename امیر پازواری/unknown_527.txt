هَرْگَهْ کِهْ مِنِهْ دُوسْتِ دِچِشْ‌شُونِهْ خٰوابْ
نَمُونِّهْ بِهْ مِنْ طٰاقِت وُ شوُنِهْ مِهْ تٰابْ
شُوی ظِلِمٰاتْ مِجِّمِهْ دُوسْتْ رِهْ بی تٰابْ
دُوسْتْ رِهْ دیمِهْ شِهْ ورْ ظِلِمٰات وُ مَهْتٰابْ
دِچِشْ نَرْگِسِ مَسِّهْ، دِلُوئِهْ عَنّٰابْ
دِ دیمْ خُوروُمُونگْ وُ دِهوُنْ حُقِّه‌یِ نٰابْ
نَدُوّمِهْ تِنِهْ خُو وُ نَدُوّمِهْ تِهْ بٰابْ
اَنْدی دُوّمِهْ کِهْ هَسِّمِهْ تِهْ عِشْقْ بی تٰابْ

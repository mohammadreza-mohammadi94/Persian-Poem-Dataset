سِهْ چِی بِهْ دَنی کِهْ دِلْ دَرْ آوَرِهْ دَرْدْ
اَوِّلْ عِشْقِهْ کِهْ گُونِهْ‌رِهْ هٰا کِنِهْ زَرْدْ
دُویُّمْ بیکَسی، تَنِ تَنْهٰا بَوَوِهْ فَرْدْ
سِیُّمْ مِفْلِسی، گِرْدِنْ کَجْ آوَرِهْ مَرْدْ
نٰاکَسْ‌رِهْ بَتّوُمِّهْ گِتِنْ شِهْ دِلِ دَرْدْ
نٰاکِسْ دٰارْمِهْ کِهْ مِهْ غَمْ‌رِهْ هٰا کِنِهْ سَرْدْ
اَمیرْ گِنِهْ اینْ غِصِّهْ مِجِمْ دینِهْ، فَرْدْ
بٰافِکْروُ خِیٰالُ خُو هٰا کِرْدْمِهْ، بَسی دَرْدْ
